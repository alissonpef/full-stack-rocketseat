-- 1-1 (chave estrangeira é unique)
CREATE TABLE students (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL
)

CREATE TABLE student_addres (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_id INTEGER UNIQUE NOT NULL

    FOREIGN KEY (student_id) REFERENCES students(id)
)

SELECT addres.id, addres.student_id, students.name
FROM student_addres AS addres
INNER JOIN students ON students.id = addres.student_id
